~~~~~~~~

~~~~~~~~

~~~~~~~~

	list-style: none;
	margin-left: 0px;
	padding-left: 0px;
}

ul.latest_updates li {
	
	padding-left: 0px;
}


.notepad-page-content > div:not(.notepad-page-title) a, .notepad-page-content > div:not(.notepad-page-title) code {
	color: grey;
